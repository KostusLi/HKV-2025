hollow action proclaim(scroll title)
[
    confession "--- ";
    confession title;
    confession " ---";
    newleaf;
    comeback;
]

squire action calctax(squire income, squire rate)
[
    elder squire result;
    result = (income * rate) / 100;
    comeback result;
]
