action squire maxthree [squire a, squire b, squire c1]
{
    elder squire temp;
    temp = mightiness(a, b);
    temp = mightiness(temp, c1);
    comeback temp;
}
action scroll unitescrolls [scroll s1, scroll s2]
{
    elder scroll result;
    elder squire len1;
    len1 = filament(s1);
    confession len1;
    newleaf;
    result = consolidate(s1, s2);
    comeback result;
}
action hollow greet [scroll name1]
{
    elder scroll hello;
    hello = "Greetings, ";
    hello = consolidate(hello, name1);
    confession hello;
    newleaf;
    comeback;
}

temple
{
    elder squire x = 10;
    elder squire y = 25;
    elder squire z = 5;
    elder squire maxval;

    elder scroll name1 = "Sir Arthur";
    elder scroll title1 = " the Brave";
    elder scroll fulltitle;
    elder rune sym = 'a';
    confession sym;
    newleaf;
	
    elder honor equalcheck = faith;

    greet(name1);

    fulltitle = unitescrolls(name1, title1);
    confession fulltitle;
    newleaf;

    equalcheck = compare_scrolls(name1, title1);

    charge [equalcheck ! exile]
    {
        confession "Titles differ.";
	newleaf;
    }
    backup
    {
        confession "Titles are equal.";
        newleaf;
    }

    patrol [x @ y]
    {
        x = x + 5;
        confession x;
	newleaf;
    }
    maxval = maxthree(12, 44, 33);
    confession maxval;
    newleaf;
    elder squire bitor;
    elder squire bitand;
    elder squire bitnot;

    bitor = x ally z;
    confession bitor;
    newleaf;
    bitand = x bind z;
    confession bitand;
    newleaf;
    bitnot = banish x;
    confession bitnot;
    newleaf;
    elder squire t = (x + y)*42 + mightiness(x, y);

    council [x]
    {
       path 1{
           confession "Path one taken.";
	   newleaf;
       }
       path 2{
           confession "Path two taken.";
	   newleaf;
       }
       tiresome{
           confession "No matching path -- tiresome.";
	   newleaf;
       }
    }
    confession "End of demonstration.";
    newleaf;
}
